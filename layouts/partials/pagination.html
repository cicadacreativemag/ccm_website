{{- /* Determine pagination format and page context */ -}} {{- $validFormats := slice "default" "terse" }} {{- $msg1 := "When passing a map to the pagination partial, one of the elements must be named 'page', and it must be set to the context of the current page." }} {{- $msg2 := "The 'format' specified in the map passed to the pagination partial is invalid. Valid choices are: %s." }} {{- $page := . }} {{- $format := "default" }} {{- if reflect.IsMap . }} {{- with .page }} {{- $page = . }} {{- else }} {{- errorf $msg1 }} {{- end }} {{- with .format }} {{- $format = lower . }} {{- end }} {{- end }} {{- /* Ensure the format is valid and render pagination */ -}} {{- if in $validFormats $format }} {{- if gt $page.Paginator.TotalPages 1 }}
    {{- partial (printf "/pagination/%s" $format) $page }}
{{- end }} {{- else }} {{- errorf $msg2 (delimit $validFormats ", ") }} {{- end -}}
